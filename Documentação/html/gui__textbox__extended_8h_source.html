108// Draw text using font inside rectangle limits with support for text selection
109static void DrawTextRecEx(Font font, const char *text, Rectangle rec, float fontSize, float spacing, bool wordWrap, Color tint, int selectStart, int selectLength, Color selectTint, Color selectBackTint)
110{
111 int length = TextLength(text); // Total length in bytes of the text, scanned by codepoints in loop
112
113 float textOffsetY = 0; // Offset between lines (on line break '\n')
114 float textOffsetX = 0.0f; // Offset X to next character to draw
115
116 float scaleFactor = fontSize/(float)font.baseSize; // Character rectangle scaling factor
117
118 // Word/character wrapping mechanism variables
119 enum { MEASURE_STATE = 0, DRAW_STATE = 1 };
120 int state = wordWrap? MEASURE_STATE : DRAW_STATE;
121
122 int startLine = -1; // Index where to begin drawing (where a line begins)
123 int endLine = -1; // Index where to stop drawing (where a line ends)
124 int lastk = -1; // Holds last value of the character position
125
126 for (int i = 0, k = 0; i < length; i++, k++)
127 {
128 // Get next codepoint from byte string and glyph index in font
129 int codepointByteCount = 0;
130 int codepoint = GetCodepoint(&text[i], &codepointByteCount);
131 int index = GetGlyphIndex(font, codepoint);
132
133 // NOTE: Normally we exit the decoding sequence as soon as a bad byte is found (and return 0x3f)
134 // but we need to draw all of the bad bytes using the '?' symbol moving one byte
135 if (codepoint == 0x3f) codepointByteCount = 1;
136 i += (codepointByteCount - 1);
137
138 float glyphWidth = 0;
139 if (codepoint != '\n')
140 {
141 glyphWidth = (font.glyphs[index].advanceX == 0) ? font.recs[index].width*scaleFactor : font.glyphs[index].advanceX*scaleFactor;
142
143 if (i + 1 < length) glyphWidth = glyphWidth + spacing;
144 }
145
146 // NOTE: When wordWrap is ON we first measure how much of the text we can draw before going outside of the rec container
147 // We store this info in startLine and endLine, then we change states, draw the text between those two variables
148 // and change states again and again recursively until the end of the text (or until we get outside of the container).
149 // When wordWrap is OFF we don't need the measure state so we go to the drawing state immediately
150 // and begin drawing on the next line before we can get outside the container.
151 if (state == MEASURE_STATE)
152 {
153 // TODO: There are multiple types of spaces in UNICODE, maybe it's a good idea to add support for more
154 // Ref: http://jkorpela.fi/chars/spaces.html
155 if ((codepoint == ' ') || (codepoint == '\t') || (codepoint == '\n')) endLine = i;
156
157 if ((textOffsetX + glyphWidth) > rec.width)
158 {
159 endLine = (endLine < 1)? i : endLine;
160 if (i == endLine) endLine -= codepointByteCount;
161 if ((startLine + codepointByteCount) == endLine) endLine = (i - codepointByteCount);
162
163 state = !state;
164 }
165 else if ((i + 1) == length)
166 {
167 endLine = i;
168 state = !state;
169 }
170 else if (codepoint == '\n') state = !state;
171
172 if (state == DRAW_STATE)
173 {
174 textOffsetX = 0;
175 i = startLine;
176 glyphWidth = 0;
177
178 // Save character position when we switch states
179 int tmp = lastk;
180 lastk = k - 1;
181 k = tmp;
182 }
183 }
184 else
185 {
186 if (codepoint == '\n')
187 {
188 if (!wordWrap)
189 {
190 textOffsetY += (font.baseSize + font.baseSize/2)*scaleFactor;
191 textOffsetX = 0;
192 }
193 }
194 else
195 {
196 if (!wordWrap && ((textOffsetX + glyphWidth) > rec.width))
197 {
198 textOffsetY += (font.baseSize + font.baseSize/2)*scaleFactor;
199 textOffsetX = 0;
200 }
201
202 // When text overflows rectangle height limit, just stop drawing
203 if ((textOffsetY + font.baseSize*scaleFactor) > rec.height) break;
204
205 // Draw selection background
206 bool isGlyphSelected = false;
207 if ((selectStart >= 0) && (k >= selectStart) && (k < (selectStart + selectLength)))
208 {
209 DrawRectangleRec((Rectangle){ rec.x + textOffsetX - 1, rec.y + textOffsetY, glyphWidth, (float)font.baseSize*scaleFactor }, selectBackTint);
210 isGlyphSelected = true;
211 }
212
213 // Draw current character glyph
214 if ((codepoint != ' ') && (codepoint != '\t'))
215 {
216 DrawTextCodepoint(font, codepoint, (Vector2){ rec.x + textOffsetX, rec.y + textOffsetY }, fontSize, isGlyphSelected? selectTint : tint);
217 }
218 }
219
220 if (wordWrap && (i == endLine))
221 {
222 textOffsetY += (font.baseSize + font.baseSize/2)*scaleFactor;
223 textOffsetX = 0;
224 startLine = endLine;
225 endLine = -1;
226 glyphWidth = 0;
227 selectStart += lastk - k;
228 k = lastk;
229
230 state = !state;
231 }
232 }
233
234 textOffsetX += glyphWidth;
235 }
236}

343RAYGUIAPI int GuiTextBoxGetByteIndex(const char *text, int start, int from, int to)
344{
345 int i = start, k = from;
346
347 while ((text[i] != '\0') && (k < to))
348 {
349 int j = 0;
350 int letter = GetCodepoint(&text[i], &j);
351
352 if (letter == 0x3f) j = 1;
353 i += j;
354 ++k;
355 }
356
357 return i;
358}

360RAYGUIAPI int GuiTextBoxDelete(char *text, int length, bool before)
361{
362 if ((guiTextBoxState.cursor != -1) && (text != NULL))
363 {
364 int startIdx = 0, endIdx = 0;
365 if ((guiTextBoxState.select != -1) && (guiTextBoxState.select != guiTextBoxState.cursor))
366 {
367 // Delete selection
368 int start = guiTextBoxState.cursor;
369 int end = guiTextBoxState.select;
370
371 if (guiTextBoxState.cursor > guiTextBoxState.select)
372 {
373 start = guiTextBoxState.select;
374 end = guiTextBoxState.cursor;
375 }
376
377 // Convert to byte indexes
378 startIdx = GuiTextBoxGetByteIndex(text, 0, 0, start);
379 endIdx = GuiTextBoxGetByteIndex(text, 0, 0, end);
380
381 // Adjust text box state
382 guiTextBoxState.cursor = start; // Always set cursor to start of selection
383 if (guiTextBoxState.select < guiTextBoxState.start) guiTextBoxState.start = -1; // Force to recalculate on the next frame
384 }
385 else
386 {
387 if (before)
388 {
389 // Delete character before cursor
390 if (guiTextBoxState.cursor != 0)
391 {
392 endIdx = GuiTextBoxGetByteIndex(text, 0, 0, guiTextBoxState.cursor);
393 guiTextBoxState.cursor--;
394 startIdx = GuiTextBoxGetByteIndex(text, 0, 0, guiTextBoxState.cursor);
395
396 if (guiTextBoxState.cursor < guiTextBoxState.start) guiTextBoxState.start = -1; // Force to recalculate on the next frame
397 }
398 }
399 else
400 {
401 // Delete character after cursor
402 if (guiTextBoxState.cursor + 1 <= GuiCountCodepointsUntilNewline(text))
403 {
404 startIdx = GuiTextBoxGetByteIndex(text, 0, 0, guiTextBoxState.cursor);
405 endIdx = GuiTextBoxGetByteIndex(text, 0, 0, guiTextBoxState.cursor+1);
406 }
407 }
408 }
409
410 memmove(&text[startIdx], &text[endIdx], length - endIdx);
411 text[length - (endIdx - startIdx)] = '\0';
412 guiTextBoxState.select = -1; // Always deselect
413
414 return (endIdx - startIdx);
415 }
416
417 return 0;
418}

460// Paste text from clipboard into the active textbox.
461// `text` is the pointer to the buffer used by the textbox while `textSize` is the text buffer max size
462RAYGUIAPI void GuiTextBoxPaste(char *text, int textSize)
463{
464 const char *clipText = GetClipboardText(); // GLFW guaratees this should be UTF8 encoded!
465 int length = strlen(text);
466
467 if ((text != NULL) && (clipText != NULL) && (guiTextBoxState.cursor != -1))
468 {
469 if ((guiTextBoxState.select != -1) && (guiTextBoxState.select != guiTextBoxState.cursor))
470 {
471 // If there's a selection we'll have to delete it first
472 length -= GuiTextBoxDelete(text, length, true);
473 }
474
475 int clipLen = strlen(clipText); // We want the length in bytes
476
477 // Calculate how many bytes can we copy from clipboard text before we run out of space
478 int size = ((length + clipLen) <= textSize) ? clipLen : textSize - length;
479
480 // Make room by shifting to right the bytes after cursor
481 int startIdx = GuiTextBoxGetByteIndex(text, 0, 0, guiTextBoxState.cursor);
482 int endIdx = startIdx + size;
483 memmove(&text[endIdx], &text[startIdx], length - startIdx);
484 text[length + size] = '\0'; // Set the NULL char
485
486 // At long last copy the clipboard text
487 memcpy(&text[startIdx], clipText, size);
488
489 // Set cursor position at the end of the pasted text
490 guiTextBoxState.cursor = 0;
491
492 for (int i = 0; i < (startIdx + size); guiTextBoxState.cursor++)
493 {
494 int next = 0;
495 int letter = GetCodepoint(&text[i], &next);
496 if (letter != 0x3f) i += next;
497 else i += 1;
498 }
499
500 guiTextBoxState.start = -1; // Force to recalculate on the next frame
501 }
502}

504RAYGUIAPI void GuiTextBoxCut(char* text)
505{
506 if ((text != NULL) &&
507 (guiTextBoxState.select != -1) &&
508 (guiTextBoxState.cursor != -1) &&
509 (guiTextBoxState.select != guiTextBoxState.cursor))
510 {
511 // First copy selection to clipboard;
512 int start = guiTextBoxState.cursor, end = guiTextBoxState.select;
513
514 if (guiTextBoxState.cursor > guiTextBoxState.select)
515 {
516 start = guiTextBoxState.select;
517 end = guiTextBoxState.cursor;
518 }
519
520 // Convert to byte indexes
521 int startIdx = GuiTextBoxGetByteIndex(text, 0, 0, start);
522 int endIdx = GuiTextBoxGetByteIndex(text, 0, 0, end);
523
524 // FIXME: `TextSubtext()` only lets use copy TEXTSPLIT_MAX_TEXT_LENGTH (1024) bytes
525 // maybe modify `SetClipboardText()` so we can use it only on parts of a string
526 const char *clipText = TextSubtext(text, startIdx, endIdx - startIdx);
527 SetClipboardText(clipText);
528
529 // Now delete selection (copy data over it)
530 int len = strlen(text);
531 memmove(&text[startIdx], &text[endIdx], len - endIdx);
532 text[len - (endIdx - startIdx)] = '\0';
533
534 // Adjust text box state
535 guiTextBoxState.cursor = start; // Always set cursor to start of selection
536 if (guiTextBoxState.select < guiTextBoxState.start) guiTextBoxState.start = -1; // Force to recalculate
537 guiTextBoxState.select = -1; // Deselect
538 }
539}

541// A text box control supporting text selection, cursor positioning and commonly used keyboard shortcuts.
542// NOTE 1: Requires static variables: framesCounter
543// NOTE 2: Returns if KEY_ENTER pressed (useful for data validation)
544RAYGUIAPI bool GuiTextBoxEx(Rectangle bounds, char *text, int textSize, bool editMode)
545{
546 // Define the cursor movement/selection speed when movement keys are held/pressed
547 #define TEXTBOX_CURSOR_COOLDOWN 5
548
549 static int framesCounter = 0; // Required for blinking cursor
550
551 GuiState state = guiState;
552 bool pressed = false;
553
554 // Make sure length doesn't exceed `textSize`. `textSize` is actually the max amount of characters the textbox can handle.
555 int length = strlen(text);
556 if (length > textSize)
557 {
558 text[textSize] = '\0';
559 length = textSize;
560 }
561
562 // Make sure we have enough room to draw at least 1 character
563 if ((bounds.width - 2*GuiGetStyle(TEXTBOX, TEXT_INNER_PADDING)) < GuiGetStyle(DEFAULT, TEXT_SIZE))
564 {
565 bounds.width = GuiGetStyle(DEFAULT, TEXT_SIZE) + 2*GuiGetStyle(TEXTBOX, TEXT_INNER_PADDING);
566 }
567
568 // Center the text vertically
569 int verticalPadding = (bounds.height - 2*GuiGetStyle(TEXTBOX, BORDER_WIDTH) - GuiGetStyle(DEFAULT, TEXT_SIZE))/2;
570
571 if (verticalPadding < 0)
572 {
573 // Make sure the height is sufficient
574 bounds.height = 2*GuiGetStyle(TEXTBOX, BORDER_WIDTH) + GuiGetStyle(DEFAULT, TEXT_SIZE);
575 verticalPadding = 0;
576 }
577
578 // Calculate the drawing area for the text inside the control `bounds`
579 Rectangle textRec = { bounds.x + GuiGetStyle(TEXTBOX, BORDER_WIDTH) + GuiGetStyle(TEXTBOX, TEXT_INNER_PADDING),
580 bounds.y + verticalPadding + GuiGetStyle(TEXTBOX, BORDER_WIDTH),
581 bounds.width - 2*(GuiGetStyle(TEXTBOX, TEXT_INNER_PADDING) + GuiGetStyle(TEXTBOX, BORDER_WIDTH)),
582 (float)GuiGetStyle(DEFAULT, TEXT_SIZE) };
583
584 Vector2 cursorPos = { textRec.x, textRec.y }; // This holds the coordinates inside textRec of the cursor at current position and will be recalculated later
585 bool active = GuiTextBoxIsActive(bounds); // Check if this textbox is the global active textbox
586
587 int selStart = 0, selLength = 0, textStartIndex = 0;
588
589 // Update control
590 //--------------------------------------------------------------------
591 if ((state != STATE_DISABLED) && !guiLocked)
592 {
593 Vector2 mousePoint = GetMousePosition();
594
595 if (editMode)
596 {
597 // Check if we are the global active textbox
598 // A textbox becomes active when the user clicks it :)
599 if (!active)
600 {
601 if (CheckCollisionPointRec(mousePoint, bounds) && (IsMouseButtonPressed(MOUSE_LEFT_BUTTON) || IsMouseButtonPressed(MOUSE_RIGHT_BUTTON)))
602 {
603 // Hurray!!! we just became the active textbox
604 active = true;
605 GuiTextBoxSetActive(bounds);
606 }
607 }
608 else if (!CheckCollisionPointRec(mousePoint, bounds) && IsMouseButtonPressed(MOUSE_RIGHT_BUTTON))
609 {
610 // When active and the right mouse is clicked outside the textbox we should deactivate it
611 // NOTE: We set a dummy rect as the active textbox bounds
612 GuiTextBoxSetActive(RAYGUI_CLITERAL(Rectangle){ 0, 0, -1, -1 });
613 active = false;
614 }
615
616 if (active)
617 {
618 state = STATE_PRESSED;
619 framesCounter++;
620
621 // Make sure state doesn't have invalid values
622 if (guiTextBoxState.cursor > length) guiTextBoxState.cursor = -1;
623 if (guiTextBoxState.select > length) guiTextBoxState.select = -1;
624 if (guiTextBoxState.start > length) guiTextBoxState.start = -1;
625
626
627 // Check textbox state for changes and recalculate if necesary
628 if (guiTextBoxState.cursor == -1)
629 {

982static int GetPrevCodepoint(const char *text, const char *start, int *prev)
983{
984 int c = 0x3f;
985 char *p = (char *)text;
986 *prev = 1;
987
988 for (int i = 0; (p >= start) && (i < 4); p--, i++)
989 {
990 if ((((unsigned char)*p) >> 6) != 2)
991 {
992 c = GetCodepoint(p, prev);
993 break;
994 }
995 }
996
997 return c;

1000// Returns total number of characters(codepoints) in a UTF8 encoded `text` until `\0` or a `\n` is found.
1001// NOTE: If a invalid UTF8 sequence is encountered a `?`(0x3f) codepoint is counted instead.
1002static inline unsigned int GuiCountCodepointsUntilNewline(const char *text)
1003{
1004 unsigned int len = 0;
1005 char *ptr = (char*)&text[0];
1006
1007 while ((*ptr != '\0') && (*ptr != '\n'))
1008 {
1009 int next = 0;
1010 int letter = GetCodepoint(ptr, &next);
1011
1012 if (letter == 0x3f) ptr += 1;
1013 else ptr += next;
1014 ++len;
1015 }
1016
1017 return len;
1018}

1020// Highly synchronized with calculations in DrawTextBoxedSelectable()
1021static int GuiMeasureTextBox(const char *text, int length, Rectangle rec, int *pos, int mode)
1022{
1023 // Get gui font properties
1024 const Font font = guiFont;
1025 const float fontSize = GuiGetStyle(DEFAULT, TEXT_SIZE);
1026 const float spacing = GuiGetStyle(DEFAULT, TEXT_SPACING);
1027
1028 int textOffsetX = 0; // Offset between characters
1029 float scaleFactor = 0.0f;
1030
1031 int letter = 0; // Current character
1032 int index = 0; // Index position in sprite font
1033
1034 scaleFactor = fontSize/font.baseSize;
1035
1036 int i = 0, k = 0;
1037 int glyphWidth = 0;
1038
1039 for (i = 0; i < length; i++, k++)
1040 {
1041 glyphWidth = 0;
1042 int next = 1;
1043 letter = GetCodepoint(&text[i], &next);
1044 if (letter == 0x3f) next = 1;
1045 index = GetGlyphIndex(font, letter);
1046 i += next - 1;
1047
1048 if (letter != '\n')
1049 {
1050 glyphWidth = (font.glyphs[index].advanceX == 0)?
1051 (int)(font.recs[index].width*scaleFactor + spacing):
1052 (int)(font.glyphs[index].advanceX*scaleFactor + spacing);
1053
1054 if ((textOffsetX + glyphWidth + 1) >= rec.width) break;
1055
1056 if ((mode == GUI_MEASURE_MODE_CURSOR_POS) && (*pos == k)) break;
1057 else if (mode == GUI_MEASURE_MODE_CURSOR_COORDS)
1058 {
1059 // Check if the mouse pointer is inside the glyph rect
1060 Rectangle grec = {rec.x + textOffsetX - 1, rec.y, (float)glyphWidth, (font.baseSize + font.baseSize/2)*scaleFactor - 1 };
1061 Vector2 mouse = GetMousePosition();
1062
1063 if (CheckCollisionPointRec(mouse, grec))
1064 {
1065 // Smooth selection by dividing the glyph rectangle into 2 equal parts and checking where the mouse resides
1066 if (mouse.x > (grec.x + glyphWidth/2))
1067 {
1068 textOffsetX += glyphWidth;
1069 k++;
1070 }
1071
1072 break;
1073 }
1074 }
1075 }
1076 else break;
1077
1078 textOffsetX += glyphWidth;
1079 }
1080
1081 *pos = k;
1082
1083 return (rec.x + textOffsetX - 1);
1084}

1183static int EncodeCodepoint(unsigned int c, char out[5])
1184{
1185 int len = 0;
1186 if (c <= 0x7f)
1187 {
1188 out[0] = (char)c;
1189 len = 1;
1190 }
1191 else if (c <= 0x7ff)
1192 {
1193 out[0] = (char)(((c >> 6) & 0x1f) | 0xc0);
1194 out[1] = (char)((c & 0x3f) | 0x80);
1195 len = 2;
1196 }
1197 else if (c <= 0xffff)
1198 {
1199 out[0] = (char)(((c >> 12) & 0x0f) | 0xe0);
1200 out[1] = (char)(((c >> 6) & 0x3f) | 0x80);
1201 out[2] = (char)((c & 0x3f) | 0x80);
1202 len = 3;
1203 }
1204 else if (c <= 0x10ffff)
1205 {
1206 out[0] = (char)(((c >> 18) & 0x07) | 0xf0);
1207 out[1] = (char)(((c >> 12) & 0x3f) | 0x80);
1208 out[2] = (char)(((c >> 6) & 0x3f) | 0x80);
1209 out[3] = (char)((c & 0x3f) | 0x80);
1210 len = 4;
1211 }
1212
1213 out[len] = 0;
1214 return len;
1215}
